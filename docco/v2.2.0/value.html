    The syntax "after" means "run the handler asynchronously after the event" This is useful, for example, to catch "keydown" events after the browser has updated the control (otherwise, ko.selectExtensions.readValue(this) will receive the control's value before the key event)

                var handler = valueUpdateHandler;
            if (ko.utils.stringStartsWith(eventName, "after")) {
                handler = function() { setTimeout(valueUpdateHandler, 0) };
                eventName = eventName.substring("after".length);
            }
            ko.utils.registerEventHandler(element, eventName, handler);
        });
